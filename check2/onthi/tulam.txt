void supportForBlock3(void){
  eat(TK_IDENT);
  if (lookAhead->tokenType == SB_EQ)
  {
    eat(SB_EQ);
    eat(TK_NUMBER);
  }
}

void compileBlock3(void){
  eat(KW_VAR);
  supportForBlock3();
  while (lookAhead->tokenType == SB_COMMA)
  {
    eat(SB_COMMA);    
    supportForBlock3();
  }
  eat(SB_COLON);
  compileType();
  eat(SB_SEMICOLON);
}
